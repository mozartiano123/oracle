zabbix - log switch interval.

"/* QUERY_LOG_SWITCH_INTERVAL */ 
        SELECT  MIN(AVG_SECONDS) AS SECONDS 
         FROM    ( 
                     SELECT  TRUNC(AVG(SECONDS)) AS AVG_SECONDS 
                     FROM    ( 
                                 SELECT  THREAD#, 
                                         LINHA, 
                                         ROUND((NEXT_TIME - FIRST_TIME) * 24 * 50 * 60) AS SECONDS 
                                 FROM    ( 
                                             SELECT  THREAD#, 
                                                     L.FIRST_TIME, 
                                                     LEAD(L.FIRST_TIME) OVER (PARTITION BY THREAD# ORDER BY SEQUENCE#) AS NEXT_TIME, 
                                                     ROW_NUMBER() OVER (PARTITION BY THREAD# ORDER BY SEQUENCE#) AS LINHA 
                                             FROM    GV$LOG L 
                                             WHERE   L.INST_ID = ( 
                                                                     SELECT  INSTANCE_NUMBER 
                                                                     FROM    V$INSTANCE 
                                                                 ) 
                                             AND     NOT EXISTS  ( 
                                                                     SELECT  1 
                                                                     FROM    GV$LOGFILE LF 
                                                                     WHERE   LF.INST_ID  =   L.INST_ID 
                                                                     AND     LF.GROUP#   =   L.GROUP# 
                                                                     AND     LF.TYPE     =  'STANDBY' 
                                                                 ) 
                                         ) LOG 
                                 WHERE       NEXT_TIME   IS  NOT NULL 
                                 AND         LINHA       <=  3 
                                 ORDER BY    THREAD#, FIRST_TIME DESC 
                             ) 
                     GROUP BY THREAD# 
                 )
